绘制单雷达图

绘制脚本如下:

var featureCompareChart = echarts.init(document.getElementById('feature_compare'));
featureCompareChart.setOption(simple_echarts.radar.radarOption([[57433, 13999, 2437, 2.56], [30832, 5569,1806, 1.78], [15769, 1378,874, 1.37]],
        ['A类客户', 'B类客户', 'C类客户'], ['用户数量' ,'收入贡献' ,'人均消费总额', '人均消费次数'], [190000,23000,2600,3],
        {title: '不同用户特征', subtitle: '2015年', title_x: 'left'}
));
