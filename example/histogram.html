绘制直方图

绘制脚本如下:

var freqChart = echarts.init(document.getElementById('frequency'));
freqChart.setOption(simple_echarts.hist.histOption(['1次', '2次', '3次及以上'], [[71.96,19.86,8.18], [86.79,11.32,1.89]],
        ['A类店', 'B类店']));

绘制带时间轴的直方图

绘制脚本如下:

var yearlyFreqChart = echarts.init(document.getElementById('yearly_frequency'));
yearlyFreqChart.setOption(simple_echarts.hist.histTimelineOption([['1次', '2次', '3次及以上'], ['1次', '2次', '3次及以上']], [[[71.96,19.86,8.18], [86.79,11.32,1.89]], [[51.96,29.86,18.18], [76.79,16.32,6.89]]],
        [['A类店', 'B类店'],['A类店', 'B类店']], ['201501', '201502'], {'title': '不同店铺类型客户购买次数分布', subtitle: '2015年', markLine: false}));
simple_echarts.resize(yearlyFreqChart);
