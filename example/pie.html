饼图

绘制脚本如下:

var salesChart = echarts.init(document.getElementById('brand_sales_pie'));
salesChart.setOption(simple_echarts.pie.pieOption([3089930773, 771758844, 640399126, 529410370], ["A", "B", "C", "D"],
        {title: '某企业各品牌销售额', subtitle: '2015年'}));

绘制环形图

绘制脚本如下:

var tableCleanChart = echarts.init(document.getElementById('table_clean'));
tableCleanChart.setOption(simple_echarts.pie.ringOption([[11, 176], [10, 716]], ['A', 'B'], [['A', 'B'], ['A', 'B']],
    {radius: ['40%', '50%'], title: 'A,B在不同情况下的占比', subtitle: '2015年', xcenters: ['30%', '70%'], ycenter: '50%'}));
