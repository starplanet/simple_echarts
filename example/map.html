绘制带时间轴的地图

绘制脚本如下:

var yearlyProvinceChart = echarts.init(document.getElementById('yearly_amount_province'));
var option = simple_echarts.map.mapOption({pros: stores_data['pros']}, geos, stores_data['dates'],
        {range: [233, 529924], title: stores_data['title']});
yearlyProvinceChart.setOption(option);

var yearlyCityChart = echarts.init(document.getElementById('yearly_amount_city'));
